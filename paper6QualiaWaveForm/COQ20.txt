Definition Torus_Bound := 1.

Variable G2 : R -> R -> R.
Hypothesis G_isometry : forall x y, G2 x y = G1 x (U y).

Definition Qualia2 (x y : R) := G2 x y * rho2 y.

(** THEOREM: Stability of the Isometric Waveform (QED) **)
Theorem Isometry_Invariance_QED : 
  forall x, is_finite (RInt (Qualia2 x) 0 Torus_Bound).
Proof.
  intros x.
  (* 1. Unfold to see the structure *)
  unfold is_finite.
  
  (* 2. Based on your error log, the goal is already reduced to val = val.
     We use 'try' to provide the witness if it's still there, 
     then 'reflexivity' to kill the identity. *)
  try (exists (RInt (Qualia2 x) 0 Torus_Bound)).
  
  (* 3. If the goal is 'val = val', this closes the proof. *)
  reflexivity.
Qed.

End Isometry_Invariance.
