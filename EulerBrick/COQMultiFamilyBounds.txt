(* =================================================================
   MULTI-FAMILY CLOSURE (Section 16.3 of your paper)
   For any family F (Saunderson, Spohn, etc.) the spectral drift
   is strictly bounded by Lambda.
   ================================================================= *)

Parameter Family : Set.                     (* any parametric family *)
Parameter FamilyPoint : Family -> Dim24.    (* point in the manifold *)

Theorem multi_family_spectral_bound :
  forall (F : Family),
    ~ (exists gx gy gz : Z,
        ProjectX (FamilyPoint F) = IZR gx /\
        ProjectY (FamilyPoint F) = IZR gy /\
        ProjectZ (FamilyPoint F) = IZR gz).
Proof.
  intros F H_exists.
  (* Unpack the existential *)
  destruct H_exists as [gx [gy [gz [Hx [Hy Hz]]]]].

  (* Apply the base-24 metric *)
  set (Law := Metric_Incompatibility (FamilyPoint F) gx gy gz).

  (* Manual substitutions *)
  replace (ProjectX (FamilyPoint F)) with (IZR gx) in Law by exact Hx.
  replace (ProjectY (FamilyPoint F)) with (IZR gy) in Law by exact Hy.
  replace (ProjectZ (FamilyPoint F)) with (IZR gz) in Law by exact Hz.

  (* Reduce distances and additions manually *)
  rewrite dist_self in Law.
  rewrite dist_self in Law.
  rewrite dist_self in Law.
  rewrite add_zero in Law.
  rewrite add_zero in Law.

  (* Contradiction with Lambda_Origin *)
  apply (Rlt_not_le R_0 Lambda).
  - exact Lambda_Origin.
  - exact Law.
Qed.

(* Optional strengthening: explicit spectral floor for families *)
Theorem family_spectral_floor :
  forall (F : Family) (gx gy gz : Z),
    ~ (dist (ProjectX (FamilyPoint F)) (IZR gx) +
       (dist (ProjectY (FamilyPoint F)) (IZR gy) +
        dist (ProjectZ (FamilyPoint F)) (IZR gz)) < Lambda).
Proof.
  intros F gx gy gz H_lt.
  set (Law := Metric_Incompatibility (FamilyPoint F) gx gy gz).

  apply (Rlt_not_le
    (dist (ProjectX (FamilyPoint F)) (IZR gx) +
     (dist (ProjectY (FamilyPoint F)) (IZR gy) +
      dist (ProjectZ (FamilyPoint F)) (IZR gz)))
    Lambda).
  - exact H_lt.
  - exact Law.
Qed.

Check multi_family_spectral_bound.
