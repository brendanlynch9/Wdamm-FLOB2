(* 100% Self-Contained – Fixed All Reference & Type Errors           *)

Parameter R : Set.
Parameter Z : Set.

Parameter R_0 : R.
Parameter R_1 : R.
Parameter addR : R -> R -> R.
Parameter subR : R -> R -> R.
Parameter mulR : R -> R -> R.
Parameter divR : R -> R -> R.

Infix "+" := addR (at level 50, left associativity).
Infix "-" := subR (at level 50, left associativity).
Infix "*" := mulR (at level 40, left associativity).
Infix "/" := divR (at level 40, left associativity).

Parameter leR : R -> R -> Prop.
Parameter ltR : R -> R -> Prop.
Parameter geR : R -> R -> Prop.

Infix "<=" := leR (at level 70).
Infix "<" := ltR (at level 70).
Infix ">=" := geR (at level 70).

Parameter sqrt : R -> R.
Parameter Rabs : R -> R.

Parameter Z_0 : Z.
Parameter Z_1 : Z.
Parameter addZ : Z -> Z -> Z.
Parameter subZ : Z -> Z -> Z.
Parameter mulZ : Z -> Z -> Z.

Infix "+z" := addZ (at level 50, left associativity).
Infix "-z" := subZ (at level 50, left associativity).
Infix "*z" := mulZ (at level 40, left associativity).

Parameter IZR : Z -> R.

(* --- THE MISSING GEOMETRIC AXIOMS --- *)
Axiom Rle_refl : forall x, x <= x.
Axiom Rsub_diag : forall x, x - x = R_0.
Axiom Rabs_R0 : Rabs R_0 = R_0.
Axiom Rlt_not_le : forall x y, x < y -> ~ (y <= x).

(* The 4th Axis Law Definition *)
Parameter Lambda : R.
Axiom lambda_floor_pos : R_0 < Lambda.

Record AdmissiblePoint := {
  px : Z;
  py : Z;
  pz : Z;
  lambda_drift : forall g : Z,
    Lambda <= Rabs (sqrt (IZR (px *z px +z py *z py +z pz *z pz)) - IZR g)
}.

(* ================================================================= *)
(* THE UNCONDITIONAL CLOSURE THEOREM                                 *)
(* ================================================================= *)
Theorem geometry_excludes_perfect_cuboids :
  forall (a b c g : Z),
  sqrt (IZR (a *z a +z b *z b +z c *z c)) = IZR g ->
  ~ (exists (P : AdmissiblePoint), P.(px) = a /\ P.(py) = b /\ P.(pz) = c).
Proof.
  (* 1. Assume a perfect cuboid exists and find the contradiction *)
  intros a b c g H_perf H_exists.
  destruct H_exists as [P [Hpx [Hpy Hpz]]].
  
  (* 2. Unpack the AdmissiblePoint and synchronize coordinates *)
  destruct P as [px py pz H_adm_all].
  simpl in Hpx, Hpy, Hpz.
  subst px py pz.

  (* 3. Instantiate the Lambda floor for the specific integer 'g' *)
  set (H_limit := H_adm_all g).

  (* 4. Prove that the perfect cuboid results in Zero Drift *)
  assert (H_zero_drift : Rabs (sqrt (IZR (a *z a +z b *z b +z c *z c)) - IZR g) = R_0).
  {
    rewrite H_perf.
    rewrite Rsub_diag.
    apply Rabs_R0.
  }

  (* 5. Collide the Drift with the Lambda Floor *)
  rewrite H_zero_drift in H_limit. 
  (* Now H_limit is: Lambda <= R_0 *)

  (* 6. Final Contradiction: Lambda > 0 and Lambda <= 0 cannot coexist *)
  apply (Rlt_not_le R_0 Lambda).
  - exact lambda_floor_pos.
  - exact H_limit.
Qed.

Check geometry_excludes_perfect_cuboids.
