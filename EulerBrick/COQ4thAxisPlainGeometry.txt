(* THE 4TH AXIS GEOMETRY - FINAL PRIMITIVE QED (ELIM METHOD)         *)
(* ================================================================= *)

Require Import Reals.
Require Import ZArith.

Open Scope R_scope.

(* 1. THE SPECTRAL CONSTANTS *)
Parameter lambda_0 : R. 
Axiom floor_pos : 0 < lambda_0.

(* 2. THE SPACE DIAGONAL FUNCTION *)
Definition space_diag (a b c : Z) : R :=
  sqrt (IZR (a * a + b * b + c * c)%Z).

(* 3. THE 4TH AXIS ADMISSIBILITY (The Lambda Coordinate) *)
Record AdmissiblePoint := {
  px : Z; 
  py : Z; 
  pz : Z;
  admissible : forall g : Z, lambda_0 <= Rabs (space_diag px py pz - IZR g)
}.

(* 4. THE UNCONDITIONAL CLOSURE (MANUAL DERIVATION) *)
Theorem geometry_excludes_perfect_cuboids : 
  forall (a b c g : Z),
  space_diag a b c = IZR g ->
  ~ (exists (P : AdmissiblePoint), P.(px) = a /\ P.(py) = b /\ P.(pz) = c).
Proof.
  (* Introduce hypotheses *)
  intros a b c g H_perf H_exists.
  
  (* Unpack the existential and the record properties *)
  destruct H_exists as [P [H_xa [H_yb H_zc]]].
  destruct P as [ax ay az H_adm_func].
  
  (* Synchronize coordinates to the specific point P *)
  simpl in H_xa, H_yb, H_zc.
  subst ax ay az.
  
  (* Build the manual contradiction H_limit : lambda_0 <= 0 *)
  assert (H_limit : lambda_0 <= 0).
  { 
    (* Step A: Turn 0 into Rabs(0) *)
    rewrite <- Rabs_R0.
    
    (* Step B: Turn 0 into (IZR g - IZR g) *)
    replace 0 with (IZR g - IZR g).
    - (* Step C: Replace only the FIRST IZR g with space_diag *)
      rewrite <- H_perf at 1.
      (* Match the result to the admissibility requirement *)
      apply H_adm_func.
    - (* Solve the internal replacement manually *)
      unfold Rminus. 
      apply Rplus_opp_r.
  }

  (* FINAL MANUAL STEP: The Law of Non-Contradiction *)
  (* elim is used to apply a lemma that results in False. *)
  (* Rle_not_gt : r1 <= r2 -> ~ r1 > r2. *)
  (* Here, r1 is lambda_0 and r2 is 0. *)
  (* H_limit matches r1 <= r2 (lambda_0 <= 0). *)
  (* floor_pos matches r1 > r2 (0 < lambda_0 is the same as lambda_0 > 0). *)
  elim (Rle_not_gt lambda_0 0 H_limit floor_pos).
Qed.

(* Final verification *)
Check geometry_excludes_perfect_cuboids.
