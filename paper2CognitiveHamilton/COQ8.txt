(* 1. FOUNDATIONAL TYPES *)
Inductive nat_manual : Type :=
| O_manual : nat_manual
| S_manual : nat_manual -> nat_manual.

Inductive list_manual (A : Type) : Type :=
| nil_manual : list_manual A
| cons_manual : A -> list_manual A -> list_manual A.

Arguments nil_manual {A}.
Arguments cons_manual {A} _ _.

(* 2. ARITHMETIC HIERARCHY *)
Fixpoint add_manual (n m : nat_manual) : nat_manual :=
  match n with | O_manual => m | S_manual n' => S_manual (add_manual n' m) end.

Fixpoint sub_manual (n m : nat_manual) : nat_manual :=
  match n, m with | O_manual, _ => O_manual | S_manual n', O_manual => n 
  | S_manual n', S_manual m' => sub_manual n' m' end.

Fixpoint mul_manual (n m : nat_manual) : nat_manual :=
  match n with | O_manual => O_manual | S_manual n' => add_manual m (mul_manual n' m) end.

Fixpoint le_manual (n m : nat_manual) : bool :=
  match n with | O_manual => true | S_manual n' =>
  match m with | O_manual => false | S_manual m' => le_manual n' m' end end.

Fixpoint eq_manual (n m : nat_manual) : bool :=
  match n with | O_manual => match m with | O_manual => true | _ => false end
  | S_manual n' => match m with | O_manual => false | S_manual m' => eq_manual n' m' end end.
