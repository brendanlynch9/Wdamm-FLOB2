Require Import Reals.
Require Import Lra.
Require Import Rfunctions.

Open Scope R_scope.

Section Classical_Unconditional_Closure.

Variable Delta : R.
Hypothesis Delta_pos : Delta > 0.

Definition Indicator_OffAxis (re_s : R) : R :=
  if (Req_EM_T re_s (1/2)) then 0 else 1.

Definition Complexity (re_s : R) : R :=
  (re_s - 1/2)^2 + Delta * (Indicator_OffAxis re_s).

Theorem Stability_Lemma :
  forall re_s : R, Complexity re_s < Delta -> re_s = 1/2.
Proof.
  intros re_s H_comp.
  unfold Complexity, Indicator_OffAxis in H_comp.
  
  destruct (Req_EM_T re_s (1/2)) as [Heq | Hneq].
  - exact Heq.
  - (* The contradiction case: re_s <> 1/2 *)
    (* We use the fact that squares are non-negative to show 
       that (sq) + Delta < Delta is impossible for Delta > 0. *)
    pose proof (pow2_ge_0 (re_s - 1/2)) as Hsq.
    lra. 
Qed.

End Classical_Unconditional_Closure.
