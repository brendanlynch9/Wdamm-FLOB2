(** --- AXIOMATIC FOUNDATIONS --- *)

Parameter carrier_type : Type.

(* Identity of Indiscernibles using Logical Bi-implication *)
Record MetricSpace := {
  carrier : Type := carrier_type;
  dist : carrier -> carrier -> R;
  dist_refl : forall x y, dist x y = 0 <-> x = y;
  dist_sym : forall x y, dist x y = dist y x;
  dist_tri : forall x y z, dist x z <= dist x y + dist y z
}.

Parameter spectral_capacity : MetricSpace -> R -> R.

(** --- SECTION 1: SPECTRAL QUOTIENT (M_delta) --- *)

Section SpectralQuotient.

Variable M : MetricSpace.
Variable delta : R.
Hypothesis delta_pos : delta > 0.

(* A delta-separated set represents the Anti-Collision Identity (ACI) *)
Definition is_delta_separated (S : carrier M -> Prop) :=
  forall x y, S x -> S y -> x <> y -> 
  spectral_capacity M (dist M x y) >= delta.

Variable S_net : carrier M -> Prop.
Hypothesis S_net_separated : is_delta_separated S_net.

Parameter pi : carrier M -> carrier M.
Hypothesis pi_in_net : forall x, S_net (pi x).

Definition M_delta (x : carrier M) : Prop := 
  exists y, pi y = x.

Theorem M_delta_is_separated :
  forall x y,
  M_delta x -> M_delta y -> x <> y ->
  spectral_capacity M (dist M x y) >= delta.
Proof.
  intros x y Hx Hy Hneq.
  destruct Hx as [x0 Hx_pi].
  destruct Hy as [y0 Hy_pi].
  subst x y.
  apply S_net_separated; try apply pi_in_net.
  exact Hneq.
Qed.

End SpectralQuotient.

(** --- SECTION 2: QUASI-ISOMETRIC STABILITY --- *)

Theorem quasi_isometric_stability :
  forall (M : Type) (dist : M -> M -> R) (delta : R) (Net : M -> Prop) (pi2 : M -> M),
  (forall x y z, dist x z <= dist x y + dist y z) -> 
  (forall x y, dist x y = dist y x) ->
  delta > 0 ->
  (forall x, Net (pi2 x) /\ dist x (pi2 x) < delta) -> 
  forall (x y : M),
  dist x y <= dist (pi2 x) (pi2 y) + 2 * delta.
Proof.
  intros M' d' delta Net pi2 Htri Hsym Hdel Hpi2 x y.
  
  (* Grounded Path: x -> pi2 x -> pi2 y -> y *)
  apply Rle_trans with (d' x (pi2 x) + d' (pi2 x) y).
  { apply Htri. }
  
  assert (H_inner: d' (pi2 x) y <= d' (pi2 x) (pi2 y) + d' (pi2 y) y) by apply Htri.
  destruct (Hpi2 x) as [_ Dx].
  destruct (Hpi2 y) as [_ Dy].
  
  (* Adjust for symmetry in the last leg of the journey *)
  rewrite Hsym in Dy.
  
  lra.
Qed.
