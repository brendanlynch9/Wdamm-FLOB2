    Verification of RKHS Monotonicity, Quotients, & Stability

Require Import Reals.
Require Import Lra.
Open Scope R_scope.

(** ====================== LEMMA 1: RKHS MONOTONICITY ====================== *)
Parameter sigma : R.
Axiom sigma_pos : sigma > 0.

Definition d_H_sq (r : R) : R := 2 * (1 - exp (- (r ^ 2) / (2 * sigma ^ 2))).

Theorem RKHS_distance_strictly_increasing :
  forall r1 r2 : R,
  0 <= r1 -> r1 < r2 ->
  sqrt (d_H_sq r1) < sqrt (d_H_sq r2).
Proof.
  intros r1 r2 H0 H12.
  apply sqrt_lt_1_alt.
  split.
  - unfold d_H_sq. apply Rmult_le_pos; [lra | ].
    rewrite <- Rminus_le_0.
    replace (exp (- (r1 ^ 2) / (2 * sigma ^ 2))) with (exp (-(r1^2 / (2 * sigma^2)))) by (field; nra).
    apply exp_le_1. nra.
  - unfold d_H_sq. apply Rmult_lt_compat_l; [lra | ].
    apply Rplus_lt_compat_l.
    apply Ropp_lt_gt_0_contravar.
    apply exp_increasing.
    replace (-(r1^2 / (2*sigma^2))) with (-(1/(2*sigma^2)) * r1^2) by (field; nra).
    replace (-(r2^2 / (2*sigma^2))) with (-(1/(2*sigma^2)) * r2^2) by (field; nra).
    apply Rmult_lt_gt_compat_neg_l.
    + apply Ropp_lt_gt_0_contravar. apply Rinv_0_lt_compat. nra.
    + apply pow_incr; lra.
Qed.

Print Assumptions RKHS_distance_strictly_increasing.

(** ====================== DEFINITION 2: SPECTRAL QUOTIENT ====================== *)
(* We use a Section to encapsulate the local metric parameters *)
Section SpectralQuotient.

Parameter M : Type.
Parameter dist : M -> M -> R.
Axiom dist_pos : forall x y, dist x y >= 0.
Axiom dist_sym : forall x y, dist x y = dist y x.
Axiom dist_tri : forall x y z, dist x z <= dist x y + dist y z.

Definition is_delta_separated (delta : R) (S : M -> Prop) :=
  forall x y, S x -> S y -> x <> y -> dist x y >= delta.

Definition is_maximal_net (delta : R) (S : M -> Prop) :=
  is_delta_separated delta S /\ forall x : M, exists y : M, S y /\ dist x y < delta.

Parameter pi : (M -> Prop) -> M -> M.
Hypothesis pi_property : forall delta S x, is_maximal_net delta S -> S (pi S x) /\ dist x (pi S x) < delta.

Theorem spectral_quotient_existence : 
  forall delta S, delta > 0 -> is_maximal_net delta S -> forall x, exists y, S y /\ dist x y < delta.
Proof.
  intros d S Hpos Hmax x. destruct Hmax as [_ Hbound]. apply Hbound.
Qed.

End SpectralQuotient.

Print Assumptions spectral_quotient_existence.

(** ====================== THEOREM 3: QUASI-ISOMETRIC STABILITY ====================== *)
Theorem quasi_isometric_stability :
  forall (M : Type) (dist : M -> M -> R) (delta : R) (Net1 Net2 : M -> Prop) (pi2 : M -> M),
  (forall x y z, dist x z <= dist x y + dist y z) -> (* Triangle Inequality *)
  (forall x y, dist x y = dist y x) ->             (* Symmetry *)
  delta > 0 ->
  (forall x, Net2 (pi2 x) /\ dist x (pi2 x) < delta) -> (* pi2 maps to Net2 within delta *)
  forall (x y : M),
  dist x y <= dist (pi2 x) (pi2 y) + 2 * delta.
Proof.
  intros M' d' delta Net1 Net2 pi2 Htri Hsym Hdel Hpi2 x y.
  (* dist(x,y) <= dist(x, pi2 x) + dist(pi2 x, y) *)
  eapply Rle_trans. apply Htri.
  (* dist(pi2 x, y) <= dist(pi2 x, pi2 y) + dist(pi2 y, y) *)
  assert (H_inner: d' (pi2 x) y <= d' (pi2 x) (pi2 y) + d' (pi2 y) y) by apply Htri.
  replace (2 * delta) with (delta + delta) by lra.
  (* Break down the inequalities using the pi2 property *)
  destruct (Hpi2 x) as [_ Dx].
  destruct (Hpi2 y) as [_ Dy].
  rewrite Hsym in Dy.
  lra.
Qed.

Print Assumptions quasi_isometric_stability.
