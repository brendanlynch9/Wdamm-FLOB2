Require Import QArith.
Require Import ZArith.

Open Scope Q_scope.

(* --- SECTION 1: POWER AND CONSTANTS --- *)

Definition TEN : Q := 10#1.

Fixpoint powQ (x:Q) (n:nat) : Q :=
  match n with
  | O => 1
  | S k => x * powQ x k
  end.

Definition TEN_POW_15 := powQ TEN 15.
Definition TEN_POW_16 := powQ TEN 16.

(* --- SECTION 2: THE UNIFIED MANIFOLD THEOREM --- *)

Theorem human_manifold_final_verification :
  ( (5/10) / (5 / TEN_POW_16) == TEN_POW_15 ) /\
  ( forall A1 A2 W F_ET, 
      0 < W -> 0 < F_ET -> A1 < A2 -> 
      (A1 / (W * F_ET)) < (A2 / (W * F_ET)) ) /\
  ( forall W_tot W_ind A_min E1 E2,
      0 < W_tot -> 0 < W_ind -> 0 < A_min -> 0 < E1 -> E1 < E2 ->
      ((W_tot / W_ind) * (A_min / E2)) < ((W_tot / W_ind) * (A_min / E1)) ).
Proof.
  split.
  - unfold Qdiv. compute. reflexivity.
  - split.
    + intros A1 A2 W F_ET HW HF HA.
      unfold Qdiv.
      assert (Hinv_pos: 0 < / (W * F_ET)).
      { apply Qinv_lt_0_compat. apply Qmult_lt_0_compat; assumption. }
      apply Qmult_lt_r; [exact Hinv_pos | exact HA].
    + intros W_tot W_ind A_min E1 E2 HWtot HWind HAmin HE1 HE12.
      unfold Qdiv.
      
      (* 1. Define the Constant Value and Prove its Positivity *)
      set (C := (W_tot * / W_ind) * A_min).
      assert (HC_pos: 0 < C).
      { unfold C. 
        apply Qmult_lt_0_compat.
        - apply Qmult_lt_0_compat; [exact HWtot | apply Qinv_lt_0_compat; exact HWind].
        - exact HAmin. }
      
      (* 2. Prove /E2 < /E1 manually *)
      assert (HE2: 0 < E2) by (apply Qlt_trans with E1; [exact HE1 | exact HE12]).
      assert (HinvE: / E2 < / E1).
      {
        assert (Hinv_prod_pos: 0 < / E1 * / E2).
        { apply Qmult_lt_0_compat; [apply Qinv_lt_0_compat; exact HE1 | apply Qinv_lt_0_compat; exact HE2]. }
        assert (Hstep: E1 * (/ E1 * / E2) < E2 * (/ E1 * / E2)) by (apply Qmult_lt_r; [exact Hinv_prod_pos | exact HE12]).
        assert (HL: E1 * (/ E1 * / E2) == / E2).
        { transitivity ((E1 * / E1) * / E2). apply Qmult_assoc.
          transitivity (1 * / E2). apply Qmult_comp; [apply Qmult_inv_r | reflexivity].
          intros H0; rewrite H0 in HE1; unfold Qlt in HE1; simpl in HE1; discriminate.
          apply Qmult_1_l. }
        assert (HR: E2 * (/ E1 * / E2) == / E1).
        { transitivity (E2 * (/ E2 * / E1)). apply Qmult_comp; [reflexivity | apply Qmult_comm].
          transitivity ((E2 * / E2) * / E1). apply Qmult_assoc.
          transitivity (1 * / E1). apply Qmult_comp; [apply Qmult_inv_r | reflexivity].
          intros H0; rewrite H0 in HE2; unfold Qlt in HE2; simpl in HE2; discriminate.
          apply Qmult_1_l. }
        rewrite HL, HR in Hstep. exact Hstep.
      }

      (* 3. Explicitly chain the inequalities using Qlt_trans *)
      (* We want to prove: Goal_L < Goal_R *)
      (* Step A: Prove Goal_L == C * /E2 *)
      assert (HEQ_L: (W_tot / W_ind) * (A_min / E2) == C * / E2).
      { unfold C, Qdiv. repeat rewrite Qmult_assoc. reflexivity. }
      
      (* Step B: Prove Goal_R == C * /E1 *)
      assert (HEQ_R: (W_tot / W_ind) * (A_min / E1) == C * / E1).
      { unfold C, Qdiv. repeat rewrite Qmult_assoc. reflexivity. }
      
      (* Step C: Apply Transitivity via Qlt_trans *)
      rewrite HEQ_L, HEQ_R.
      apply Qmult_lt_l; [exact HC_pos | exact HinvE].
Qed.

(* --- FINAL SYSTEM CHECK --- *)
Check human_manifold_final_verification.
