(* 1. ARITHMETIC FOUNDATION *)
Inductive nat_manual : Type :=
| O_manual : nat_manual
| S_manual : nat_manual -> nat_manual.

Fixpoint add_manual (n m : nat_manual) : nat_manual :=
  match n with
  | O_manual => m
  | S_manual n' => S_manual (add_manual n' m)
  end.

Fixpoint mul_manual (n m : nat_manual) : nat_manual :=
  match n with
  | O_manual => O_manual
  | S_manual n' => add_manual m (mul_manual n' m)
  end.

Fixpoint sub_manual (n m : nat_manual) : nat_manual :=
  match n, m with
  | O_manual, _ => O_manual
  | S_manual n', O_manual => n
  | S_manual n', S_manual m' => sub_manual n' m'
  end.

Fixpoint le_manual (n m : nat_manual) : bool :=
  match n with
  | O_manual => true
  | S_manual n' =>
      match m with
      | O_manual => false
      | S_manual m' => le_manual n' m'
      end
  end.

Record Q_manual := mkQ { num : nat_manual; den : nat_manual }.

Definition Q_le (q1 q2 : Q_manual) : bool :=
  le_manual (mul_manual q1.(num) q2.(den)) (mul_manual q2.(num) q1.(den)).

(* 3. FUELED DIVISION & CEILING *)
Fixpoint div_fueled (fuel n d : nat_manual) : nat_manual :=

  | S_manual fuel' =>

Definition ceiling_manual (n d : nat_manual) : nat_manual :=
